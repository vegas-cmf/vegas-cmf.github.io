namespace Vegas\Paginator;

/**
 * Class Page
 *
 * Represents page from paginator
 *
 * @package Vegas\Paginator
 */
class Page
{
    /**
     * Number of total pages
     *
     * @var int
     */
    public $total_pages;

    /**
     * Number of previous page
     *
     * @var int
     */
    public $before;

    /**
     * Number of current page
     *
     * @var int
     */
    public $current;

    /**
     * Number of next page
     *
     * @var int
     */
    public $next;

    /**
     * Pagination items
     *
     * @var array()|\Iterator
     */
    public $items;
}
