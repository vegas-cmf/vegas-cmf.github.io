namespace Vegas\Http\Response;

/**
 * Class Json
 * @package Vegas\Http\Response
 */
class Json implements \JsonSerializable
{
    /**
     * Determines if request was succeed or failed
     *
     * @var bool
     * @internal
     */
    private $isSuccess = false;

    /**
     * Contains data
     *
     * @var string
     * @internal
     */
    private $data = '';

    /**
     * Response message
     *
     * @var string
     * @internal
     */
    private $message = '';

    /**
     * Determines if request was succeed
     *
     * @return $this
     */
    public function success()
    {
        $this->isSuccess = true;
        return $this;
    }

    /**
     * Determines if request was failed
     *
     * @return $this
     */
    public function fail()
    {
        $this->isSuccess = false;
        return $this;
    }

    /**
     * Sets request response message
     *
     * @param $message
     * @return $this
     */
    public function setMessage($message)
    {
        $this->message = $message;
        return $this;
    }

    /**
     * Sets request data
     *
     * @param $data
     * @return $this
     */
    public function setData($data)
    {
        $this->data = $data;
        return $this;
    }

    /**
     * Serializes array to JSON format
     *
     * @return array|mixed
     */
    public function jsonSerialize()
    {
        return array(
            'success'   =>  $this->isSuccess,
            'data'      =>  $this->data,
            'message'     =>  $this->message
        );
    }
}
